### 파일 변경: `src/hooks/useObservation.js`

기존 submit 함수 안에 `updateStreak(userId)` 호출 추가.

```js
// src/hooks/useObservation.js 에 추가

function toKSTDateString(date) {
  // 'YYYY-MM-DD' in KST (UTC+9)
  return new Date(date.getTime() + 9 * 60 * 60 * 1000)
    .toISOString()
    .slice(0, 10)
}

async function updateStreak(userId) {
  const today = toKSTDateString(new Date())
  const yesterday = toKSTDateString(new Date(Date.now() - 86400000))

  const { data: profile } = await supabase
    .from('profiles')
    .select('streak_days, last_observed_date, streak_shield')
    .eq('id', userId)
    .single()

  const last = profile.last_observed_date

  if (last === today) return // 오늘 이미 기록 — 변경 없음

  let streak = profile.streak_days
  let shield = profile.streak_shield

  if (last === yesterday) {
    streak += 1
  } else if (last !== null) {
    // 하루 이상 끊김
    if (shield > 0) {
      shield -= 1
      streak += 1
    } else {
      streak = 1
    }
  } else {
    // 첫 관측
    streak = 1
  }

  await supabase
    .from('profiles')
    .update({
      streak_days: streak,
      last_observed_date: today,
      streak_shield: shield,
    })
    .eq('id', userId)
}
```

기존 submit 함수 마지막에 추가:

```js
await updateStreak(user.id)
```

## 3. 지역 기여 카운터

### 신규 훅: `src/hooks/useLocalContribution.js`

```js
// src/hooks/useLocalContribution.js
import { useEffect, useState } from 'react'
import { supabase } from '@/lib/supabase'

export function useLocalContribution(lat, lng, userId) {
  const [data, setData] = useState(null)

  useEffect(() => {
    if (!lat || !lng || !userId) return

    async function load() {
      const delta = 10 / 111 // 10km를 위도/경도 도 단위로 근사
      const since = new Date(Date.now() - 7 * 24 * 60 * 60 * 1000).toISOString()

      const { data: obs, error } = await supabase
        .from('observations')
        .select('user_id')
        .gte('lat', lat - delta)
        .lte('lat', lat + delta)
        .gte('lng', lng - delta)
        .lte('lng', lng + delta)
        .gte('observed_at', since)

      if (error || !obs) return

      const uniqueUsers = new Set(obs.map(o => o.user_id)).size
      const totalObs = obs.length
      const myObs = obs.filter(o => o.user_id === userId).length
      const myRatio = totalObs > 0 ? Math.round((myObs / totalObs) * 100) : 0

      setData({ uniqueUsers, myRatio })
    }

    load()
  }, [lat, lng, userId])

  return data
}
```

## 주의사항

- `toKSTDateString` 함수는 `src/utils/date.js`로 분리해도 무방 (streak 외에도 쓸 수 있음)
- `delta = 10 / 111`은 위도 기준 근사값. 경도는 위도에 따라 달라지지만 한국 위도(34~38°)에서 오차 약 10% 이내로 허용 가능
- `LocalContributionCard`는 위치 권한이 없으면(`location === null`) 훅이 실행되지 않아 자동으로 렌더 안 됨 — 별도 분기 불필요
- 이번 세션에서 shield 월 1회 지급 cron은 구현하지 않음 — 가입 시 기본값 1로 충분
